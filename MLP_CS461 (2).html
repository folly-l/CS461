Conceptual Question

KNN: Does classification by finding the k nearest neighbors and assigning the most common label among them.

Decision Trees: Recursively splits data based on feature thresholds that can maximize information gain, forming a tree where each leaf represents a class label.

Naïve Bayes: Uses Bayes' theorem with the assumption that features are independent, assigning the label with the highest posterior probability given the feature values.

K-Means: splits the data into k clusters by assigning each point to the nearest cluster centroid and iteratively updating centroids to minimize within-cluster variance.

class KNN:
    def __init__(self, k=3):
        """
        Initialize KNN with a given number of neighbors (k).
        """
        self.k = k

    def fit(self, X, y):
        """
        Store the training data and labels.
        KNN is a lazy learner — it doesn't train a model,
        just remembers all training examples for future distance comparisons.
        """
        self.X_train = X
        self.y_train = y

    def predict(self, X):
        """
        Predict labels for each example in X.
        For each input sample:
          1. Compute distances to all training samples
          2. Identify the k nearest neighbors
          3. Use majority vote to assign the class
        Returns an array of predictions.
        """
        predictions = [self._predict(x) for x in X]
        return np.array(predictions)

    def _predict(self, x):
        """
        Predict label for a single example.
        Steps:
          1. Compute Euclidean distance between x and every training point
          2. Sort by distance and select the top-k indices
          3. Retrieve corresponding labels
          4. Perform majority voting among those labels
        Returns the most frequent label among nearest neighbors.
        """

        # Computing distances, sorting, picking top k with labels
        eu = euclidean_distance(self.X_train, x)
        k_idx = np.argsort(eu)[:self.k]
        k_labels = self.y_train[k_idx]
        # Majority vote
        vals, counts = np.unique(k_labels, return_counts=True)
        majority_vote = vals[np.argmax(counts)]

        return majority_vote

def stratified_split(df, target_col, seed=42):
    np.random.seed(seed)
    train_df, val_df, test_df = [], [], []

    for cls, subset in df.groupby(target_col):
        subset = subset.sample(frac=1, random_state=seed)
        n = len(subset)
        n_train = int(0.7 * n)
        n_val = int(0.15 * n) # taking only n_val to avoid rounding issues

        train_df.append(subset.iloc[:n_train])
        val_df.append(subset.iloc[n_train:n_train+n_val])
        test_df.append(subset.iloc[n_train+n_val:])

    train_df = pd.concat(train_df)
    val_df = pd.concat(val_df)
    test_df = pd.concat(test_df)
    return train_df, val_df, test_df

class NaiveBayes:
    def __init__(self, epsilon=1e-6):
        self.epsilon = epsilon
        self.classes = None
        self.log_prior = {}
        self.mean = {}
        self.var = {}


    def fit(self, X, y):
        """Compute priors and likelihoods in log-space."""
        X = np.asarray(X, dtype=np.float64)  # force numeric array..due to ValueError and attributeErrors lol
        y = np.asarray(y)
        self.classes = np.unique(y)
        self._calculate_prior(y)
        self._calculate_likelihood(X, y)



    def _calculate_prior(self, y):
        """Return log-priors for each class."""
        n_samples = len(y)
        self.log_prior = {cls: np.log(np.sum(y == cls) / n_samples) for cls in self.classes}



    def _calculate_likelihood(self, X, y):
        """Return log-likelihood parameters (mean, var for Gaussian NB)."""
        for cls in self.classes:
            X_c = X[y == cls]
            self.mean[cls] = np.array(X_c.mean(axis=0), dtype=np.float64)
            self.var[cls] = np.array(X_c.var(axis=0), dtype=np.float64) + self.epsilon

    def _log_probability(self, x, cls):
        """Compute log P(class|x) using priors + likelihoods."""
        x = np.asarray(x, dtype=np.float64)
        mean = self.mean[cls]
        var = self.var[cls]
        log_likelihood = -0.5 * np.sum(np.log(2 * np.pi * var))
        log_likelihood -= 0.5 * np.sum((x - mean)**2 / var)
        return self.log_prior[cls] + log_likelihood

    def predict(self, X):
        """Predict class labels for samples."""
        X = np.asarray(X, dtype=np.float64)
        y_pred = []
        for i in range(X.shape[0]):
            log_probs = [self._log_probability(X[i], cls) for cls in self.classes]
            y_pred.append(self.classes[np.argmax(log_probs)])
        return np.array(y_pred)

import numpy as np

class KMeansFromScratch:
    def __init__(self, n_clusters=3, max_iter=300, tol=1e-4, random_state=42):
        self.n_clusters = n_clusters
        self.max_iter = max_iter
        self.tol = tol
        self.random_state = random_state
        self.centroids = None
        self.labels_ = None
        self.inertia_ = None

    def fit(self, X):
        """
        Fit KMeans model to data X.
        Steps:
        1. Initialize centroids randomly
        2. Repeat until convergence or max_iter:
           - Assign each point to nearest centroid
           - Update centroids as mean of assigned points
           - Compute inertia (WCSS)
        """
        np.random.seed(self.random_state)
        self._initialize_centroids(X)

        for i in range(self.max_iter):
            labels = self._assign_clusters(X)
            new_centroids = self._update_centroids(X, labels)
            diff = np.linalg.norm(self.centroids - new_centroids)
            self.centroids = new_centroids

            if diff < self.tol:
                break


        self.labels_ = self._assign_clusters(X)
        self.inertia_ = self._compute_inertia(X, self.labels_)

    def _initialize_centroids(self, X):
        """
        Pick initial centroids from X (randomly or k-means++).
        """
        idx = np.random.choice(X.shape[0], self.n_clusters, replace=False)
        self.centroids = X[idx]

    def _assign_clusters(self, X):
        """
        Assign each data point to the closest centroid.
        Returns: array of cluster labels
        """
        distances = np.linalg.norm(X[:, np.newaxis] - self.centroids, axis=2)

        return np.argmin(distances, axis=1)

    def _update_centroids(self, X, labels):
        """
        Compute new centroids as mean of assigned points.
        """

        centroids = np.zeros((self.n_clusters, X.shape[1]))
        for k in range(self.n_clusters):
            points = X[labels == k]
            if len(points) > 0:
                centroids[k] = points.mean(axis=0)
            else:

                centroids[k] = X[np.random.randint(0, X.shape[0])]
        return centroids

    def _compute_inertia(self, X, labels):
        """
        Compute WCSS (within-cluster sum of squares).
        """

        inertia = 0
        for k in range(self.n_clusters):
            points = X[labels == k]
            inertia += ((points - self.centroids[k])**2).sum()
        return inertia

    def predict(self, X):
        """
        Assign clusters for new unseen data based on nearest centroid.
        """

        distances = np.linalg.norm(X[:, np.newaxis] - self.centroids, axis=2)
        return np.argmin(distances, axis=1)

def manual_silhouette_score_np(X, labels):
    n_samples = X.shape[0]
    unique_labels = np.unique(labels)
    s_list = np.zeros(n_samples)

    for i in range(n_samples):
        own_cluster = labels[i]
        own_mask = labels == own_cluster
        other_mask = labels != own_cluster

        # a_i: intra-cluster distance (cohesion)
        same_cluster_points = X[own_mask]
        if same_cluster_points.shape[0] == 1:
            a_i = 0
        else:

            diffs = same_cluster_points - X[i]
            dists = np.linalg.norm(diffs, axis=1)

            a_i = np.sum(dists) / (len(dists) - 1)

        # b_i: nearest-cluster distance (separation)
        b_list = []
        for other_label in unique_labels:
            if other_label == own_cluster:
                continue
            other_points = X[labels == other_label]
            diffs = other_points - X[i]
            dists = np.linalg.norm(diffs, axis=1)
            b_list.append(np.mean(dists))
        b_i = min(b_list)

        # silhouette for point i
        s_list[i] = (b_i - a_i) / max(a_i, b_i) if max(a_i, b_i) != 0 else 0

    return np.mean(s_list)

REFLECTION¶

Why K-Means struggles with true species? The K-Means assumes spherical and equal sized clusters, but from the graph, we can see that iris species have overlapping distribution in some of its features. Due to this, some points might not be clustered properly (potentially).
